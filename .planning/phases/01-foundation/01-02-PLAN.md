    ```
    def generate_reward(
        cue_chosen: int,
        cue_probs: tuple[float, ...],
        rng: numpy.random.Generator,
    ) -> int:
        """Generate a binary reward for the chosen cue.

        Parameters
        ----------
        cue_chosen : int
            Index of the cue selected by the agent.
        cue_probs : tuple[float, ...]
            Ground truth reward probabilities for each cue.
        rng : numpy.random.Generator
            Random number generator for stochastic reward.

        Returns
        -------
        int
            1 if rewarded, 0 if not.
        """
        return int(rng.random() < cue_probs[cue_chosen])
    ```

    Define generate_session function:
    ```
    def generate_session(config: AnalysisConfig, seed: int) -> list[Trial]:
        """Generate one session's trial sequence from config.

        Iterates over sets and phases to produce a complete trial list.
        Each set contains all phases (acquisition, reversals, post-reversal)
        plus a transfer phase at the end.

        Parameters
        ----------
        config : AnalysisConfig
            Validated analysis configuration.
        seed : int
            Random seed (used for potential future stochastic elements;
            currently session structure is deterministic given config).

        Returns
        -------
        list[Trial]
            Complete ordered list of trials for one session.
        """
        # NOTE: The session structure itself is deterministic from config.
        # The seed is accepted for forward-compatibility (Phase 3 will use
        # it for stochastic reward generation).

        rng = np.random.default_rng(seed)
        trials: list[Trial] = []
        trial_idx = 0

        for set_idx in range(config.task.n_sets):
            # Main phases (acquisition + reversals + post_reversal)
            for phase in config.task.phases:
                for _ in range(phase.n_trials):
                    trials.append(Trial(
                        trial_idx=trial_idx,
                        set_idx=set_idx,
                        phase_name=phase.name,
                        phase_label=phase.phase_label,
                        cue_probs=tuple(phase.cue_probs),
                        best_cue=int(np.argmax(phase.cue_probs)),
                    ))
                    trial_idx += 1

            # Transfer phase
            transfer = config.task.transfer
            for _ in range(transfer.n_trials):
                trials.append(Trial(
                    trial_idx=trial_idx,
                    set_idx=set_idx,
                    phase_name="transfer",
                    phase_label=transfer.phase_label,
                    cue_probs=tuple(transfer.cue_probs),
                    best_cue=int(np.argmax(transfer.cue_probs)),
                ))
                trial_idx += 1

        return trials
    ```
