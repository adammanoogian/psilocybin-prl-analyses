   ```python
       params : tuple[str, ...], default ('omega_2', 'beta')

       Returns
       -------
       pd.DataFrame
           Long format: one row per (participant_id, parameter). Columns:
           participant_id, parameter, mean_laplace, mean_nuts,
           sd_laplace, sd_nuts, abs_diff_mean, abs_diff_log_sd, sd_ratio,

           informational only (no tolerance defined for them in Phase 19;

           Column dtype: ``pd.BooleanDtype()`` (nullable bool).
       """
   ```

2. `_apply_hard_gates(df: pd.DataFrame) -> tuple[bool, list[str]]`:
   Returns `(all_pass, messages)`. Hard gate operates on
   `df[df.parameter == "omega_2"]` ONLY. Equivalent to checking whether
   `within_gate.all()` is True on the omega_2-filtered frame.
   Non-omega_2 rows are NEVER consulted for gating (they retain
   `pd.NA` in `within_gate`); soft sd_ratio warnings still scan every
   row since they are informational regardless of parameter.
   ```python
   omega2 = df[df.parameter == "omega_2"]
   # within_gate on omega_2 rows is guaranteed non-null by the factory
   # contract; use .all() directly.
   all_pass = bool(omega2.within_gate.all())
   messages: list[str] = []
   for _, row in omega2.iterrows():
       if row.abs_diff_mean >= _GATE_ABS_DIFF_MEAN_OMEGA2:
           messages.append(
               f"HARD FAIL: participant={row.participant_id} "
               f"abs_diff_mean(omega_2)={row.abs_diff_mean:.3f} "
               f">= gate {_GATE_ABS_DIFF_MEAN_OMEGA2}"
           )
       if row.abs_diff_log_sd >= _GATE_ABS_DIFF_LOG_SD_OMEGA2:
           messages.append(
               f"HARD FAIL: participant={row.participant_id} "
               f"abs_diff_log_sd(omega_2)={row.abs_diff_log_sd:.3f} "
               f">= gate {_GATE_ABS_DIFF_LOG_SD_OMEGA2}"
           )
   # Soft SD-ratio warnings scan every row (all parameters).
   for _, row in df.iterrows():
       if not (_SOFT_SD_RATIO_LOWER <= row.sd_ratio <= _SOFT_SD_RATIO_UPPER):
           messages.append(
               f"SOFT WARN: participant={row.participant_id} "
               f"parameter={row.parameter} sd_ratio={row.sd_ratio:.3f} "
               f"outside [{_SOFT_SD_RATIO_LOWER}, {_SOFT_SD_RATIO_UPPER}]"
           )
   return all_pass, messages
   ```
   CLI `main()` uses `return 0 if all_pass else 1` (exit 0 when all
   omega_2 rows are within both tolerance conditions; exit 1 otherwise).

3. CLI `main()`:
   ```python
   def main() -> int:
       parser = argparse.ArgumentParser(description=__doc__)
       sub = parser.add_subparsers(dest="cmd", required=True)

       p_run = sub.add_parser(
           "run",
           help="Run Laplace fit on sim_df; optionally also run NUTS "
           "for comparison. Saves both idatas to --output-dir.",
       )
       p_run.add_argument("--sim-df", type=Path, required=True,
                          help="Path to parquet sim_df")
       p_run.add_argument("--model", type=str, default="hgf_2level_patrl")
       p_run.add_argument("--output-dir", type=Path, required=True)
       p_run.add_argument("--skip-nuts-comparison", action="store_true",
                          help="Only run Laplace; skip NUTS fit.")

       p_cmp = sub.add_parser(
           "compare",
           help="Load two saved idatas and produce a diff table.",
       )
       p_cmp.add_argument("--laplace", type=Path, required=True,
                          help="Path to Laplace idata .nc file")
       p_cmp.add_argument("--nuts", type=Path, required=True,
                          help="Path to NUTS idata .nc file")
       p_cmp.add_argument("--output", type=Path, default=None,
                          help="Optional path to write diff CSV.")
       p_cmp.add_argument("--params", type=str, nargs="+",
                          default=["omega_2", "beta"])

       args = parser.parse_args()

       if args.cmd == "run":
           ...  # see below
       else:  # compare
           idata_lap = az.from_netcdf(str(args.laplace))
           idata_nuts = az.from_netcdf(str(args.nuts))
           df = compare_posteriors(idata_lap, idata_nuts, tuple(args.params))
           if args.output is not None:
               df.to_csv(args.output, index=False)
               logger.info("Wrote diff CSV to %s", args.output)
           all_pass, messages = _apply_hard_gates(df)
           for msg in messages:
               if msg.startswith("HARD FAIL"):
                   logger.error(msg)
               else:
                   logger.warning(msg)
           return 0 if all_pass else 1

       return 0
   ```
