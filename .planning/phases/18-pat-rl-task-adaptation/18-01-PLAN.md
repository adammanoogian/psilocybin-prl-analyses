```yaml
task:
  name: pat_rl
  description: >
    Predator Avoidance Task — Reinforcement Learning variant. Binary
    safe/dangerous context state, approach/avoid action, 2x2 reward/shock
    magnitude design, hazard-driven reversals. 192 trials across 4 runs.
  n_trials: 192
  n_runs: 4
  trials_per_run: 48

  # Hazard-rate-driven reversal (Behrens 2007 / Browning 2015 pattern).
  # Two counterbalanced regimes across the 4 runs: stable/volatile.
  hazards:
    stable: 0.03
    volatile: 0.10
  run_order: ["stable", "volatile", "stable", "volatile"]

  # Outcome contingencies given state:
  # state=safe (0): P(reward | approach)=0.75, P(shock | approach)=0.00, P(nothing | approach)=0.25
  # state=dangerous (1): P(reward | approach)=0.25, P(shock | approach)=0.50, P(nothing | approach)=0.25
  # avoid action always yields nothing (no outcome).
  contingencies:
    safe:
      reward: 0.75
      shock: 0.00
      nothing: 0.25
    dangerous:
      reward: 0.25
      shock: 0.50
      nothing: 0.25

  # 2x2 reward/shock magnitude design crossed within-subject per trial.
  # Each trial drawn uniformly at random from the 4 cells.
  magnitudes:
    reward_levels: [1.0, 5.0]
    shock_levels: [1.0, 5.0]

  # Trial timing in seconds (needed for trajectory export outcome_time_s).
  timing:
    cue_duration_s: 1.5
    anticipation_s: 5.5
    outcome_duration_s: 2.0
    iti_s: 2.0
    # total per trial = 11.0 s nominal; run gap = 15.0 s

  # Delta-HR stub distribution (caller may override with real data).
  # anticipatory bradycardia: N(-3, 3) on dangerous anticipation,
  # N(0, 3) on safe anticipation; hard bounds [-15, 10] bpm.
  delta_hr_stub:
    dangerous: {mean: -3.0, sd: 3.0}
    safe: {mean: 0.0, sd: 3.0}
    bounds: [-15.0, 10.0]

simulation:
  n_participants_per_phenotype: 8   # 4 phenotypes x 8 = 32 for V1/V2
  master_seed: 5678

  # Phenotype 2x2 grid (anxiety x reward_sensitivity) per research addendum.
  # Numbers are Browning 2015 / Daw 2006 / Schönberg 2007 grounded; tune in V2.
  phenotypes:
    healthy:
      omega_2: {mean: -6.0, sd: 0.5}
      beta:    {mean: 2.0, sd: 0.5}
      kappa:   {mean: 1.0, sd: 0.0}    # fixed
      mu3_0:   {mean: 1.0, sd: 0.0}
    anxious:
      omega_2: {mean: -3.5, sd: 0.5}
      beta:    {mean: 2.0, sd: 0.5}
      kappa:   {mean: 1.0, sd: 0.0}
      mu3_0:   {mean: 1.0, sd: 0.0}
    reward_sensitive:
      omega_2: {mean: -6.0, sd: 0.5}
      beta:    {mean: 8.0, sd: 0.5}
      kappa:   {mean: 1.0, sd: 0.0}
      mu3_0:   {mean: 1.0, sd: 0.0}
    anxious_reward_sensitive:
      omega_2: {mean: -3.5, sd: 0.5}
      beta:    {mean: 8.0, sd: 0.5}
      kappa:   {mean: 1.0, sd: 0.0}
      mu3_0:   {mean: 1.0, sd: 0.0}

fitting:
  # MCMC defaults for smoke run; production values live in SLURM scripts.
  n_chains: 2
  n_tune: 500
  n_draws: 500
  target_accept: 0.90
  random_seed: 42
  priors:
    omega_2: {mean: -4.0, sd: 2.0}
    omega_3: {mean: -6.0, sd: 2.0}
```

```python
"""PAT-RL task configuration loader (parallel stack from pick_best_cue).

Loads ``configs/pat_rl.yaml`` and returns a validated, typed
:class:`PATRLConfig` hierarchy.  This loader is completely independent of
:mod:`prl_hgf.env.task_config` — it does not import, subclass, or inspect
``AnalysisConfig``/``TaskConfig``.  That keeps pick_best_cue tests isolated
from PAT-RL changes.
"""

from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path
from typing import Any

import yaml

from config import CONFIGS_DIR

_DEFAULT_PATRL_CONFIG_PATH = CONFIGS_DIR / "pat_rl.yaml"
```

Define (all `frozen=True` dataclasses, NumPy-style docstrings, `__post_init__`
validation). Use native 3.10+ type hints.

```python
@dataclass(frozen=True)
class HazardConfig:
    stable: float
    volatile: float
    # __post_init__: both in (0, 1); stable < volatile

@dataclass(frozen=True)
class OutcomeProbs:
    reward: float
    shock: float
    nothing: float
    # __post_init__: each in [0,1]; sum == 1 (atol=1e-6)

@dataclass(frozen=True)
class ContingencyConfig:
    safe: OutcomeProbs
    dangerous: OutcomeProbs

@dataclass(frozen=True)
class MagnitudeConfig:
    reward_levels: tuple[float, ...]
    shock_levels: tuple[float, ...]
    # __post_init__: each list len == 2, all > 0

@dataclass(frozen=True)
class TimingConfig:
    cue_duration_s: float
    anticipation_s: float
    outcome_duration_s: float
    iti_s: float
    # __post_init__: all > 0

    @property
    def trial_duration_s(self) -> float:
        return (self.cue_duration_s + self.anticipation_s
                + self.outcome_duration_s + self.iti_s)

@dataclass(frozen=True)
class DeltaHRDistribution:
    mean: float
    sd: float
    # __post_init__: sd > 0

@dataclass(frozen=True)
class DeltaHRStubConfig:
    dangerous: DeltaHRDistribution
    safe: DeltaHRDistribution
    bounds: tuple[float, float]
    # __post_init__: bounds[0] < bounds[1]

@dataclass(frozen=True)
class PATRLTaskConfig:
    name: str
    description: str
    n_trials: int
    n_runs: int
    trials_per_run: int
    hazards: HazardConfig
    run_order: tuple[str, ...]
    contingencies: ContingencyConfig
    magnitudes: MagnitudeConfig
    timing: TimingConfig
    delta_hr_stub: DeltaHRStubConfig
    # __post_init__: n_runs * trials_per_run == n_trials;
    # run_order length == n_runs; each entry in {"stable","volatile"}

@dataclass(frozen=True)
class PriorGaussian:
    mean: float
    sd: float
    # sd > 0

@dataclass(frozen=True)
class PriorTruncated:
    lower: float
    upper: float
    mean: float
    sd: float
    # lower < upper; sd > 0; lower <= mean <= upper

@dataclass(frozen=True)
class PhenotypeParams:
    omega_2: PriorGaussian
    beta: PriorGaussian
    kappa: PriorGaussian
    mu3_0: PriorGaussian

@dataclass(frozen=True)
class PATRLSimulationConfig:
    n_participants_per_phenotype: int
    master_seed: int
    phenotypes: dict[str, PhenotypeParams]   # keys are phenotype names
    # __post_init__: exactly the 4 keys {healthy, anxious,
    # reward_sensitive, anxious_reward_sensitive}; n_participants_per_phenotype >= 1

@dataclass(frozen=True)
class FittingPriorConfig:
    omega_2: PriorGaussian
    omega_3: PriorGaussian
    kappa: PriorTruncated
    beta: PriorTruncated
    mu3_0: PriorGaussian

@dataclass(frozen=True)
class PATRLFittingConfig:
    n_chains: int
    n_tune: int
    n_draws: int
    target_accept: float
    random_seed: int
    priors: FittingPriorConfig
    # __post_init__: n_chains >= 1, n_tune >= 1, n_draws >= 1,
    # 0 < target_accept < 1

@dataclass(frozen=True)
class PATRLConfig:
    task: PATRLTaskConfig
    simulation: PATRLSimulationConfig
    fitting: PATRLFittingConfig
```
