   ```python
   def _fit(
       sim_df: pd.DataFrame,
       level: int,
       method: str,
       n_tune: int,
       n_draws: int,
       seed: int,
       n_pseudo_draws: int = 1000,
   ) -> tuple[az.InferenceData, az.InferenceData | None]:
       """Dispatch fit by method; returns (primary_idata, optional_secondary).

       Primary idata is what downstream export consumes. For 'both',
       primary = laplace, secondary = blackjax (to reflect Phase 19
       being the "new" path under test while the blackjax path is the
       baseline).
       """
       model_name = (
           "hgf_2level_patrl" if level == 2 else "hgf_3level_patrl"
       )
       if method == "blackjax":
           idata = _fit_blackjax(sim_df, level, n_tune, n_draws, seed)
           return idata, None
       if method == "laplace":
           idata = _fit_laplace(
               sim_df, level, n_pseudo_draws=n_pseudo_draws,
               seed=seed,
           )
           return idata, None
       # both
       idata_nuts = _fit_blackjax(sim_df, level, n_tune, n_draws, seed)
       idata_lap = _fit_laplace(
           sim_df, level, n_pseudo_draws=n_pseudo_draws, seed=seed,
       )
       return idata_lap, idata_nuts
   ```

   kwarg. **FORBID:** bare `try/except` around `sample_stats` access.

   The `both` branch uses `primary` and `secondary` from Step 3's

   NUTS when `method == "both"`).

   ```python
   def _sanity_check(
       idata: az.InferenceData,
       method: str,
   ) -> None:
       """Method-specific sanity check on fit diagnostics.

       Parameters
       ----------
       idata : az.InferenceData
           Fit result to check. For ``method='both'``, pass the primary
           idata; call separately on the secondary.
       method : str
           One of ``'blackjax'``, ``'laplace'``, ``'both'``.
       """
       if method == "blackjax":
           div = idata.sample_stats.diverging
           frac = float(div.sum()) / float(div.size)
           if frac >= 0.20:
               raise RuntimeError(
                   f"NUTS divergence fraction {frac:.3f} >= 0.20 gate"
               )
       elif method == "laplace":
           if not bool(idata.sample_stats.converged.values.any()):
               logger.warning(
                   "Laplace fit did not converge for any subject. "
                   "Check diagnostics (logp_at_mode, hessian_min_eigval, "
                   "n_eigenvalues_clipped) for the failing subjects. "
                   "Continuing — Laplace may fail per-subject; this is "
                   "a soft gate at the smoke level."
               )
       elif method == "both":
           # Dispatch path for --fit-method=both. Uses the primary/
           # secondary naming from Step 3's _fit dispatch:
           # primary = Laplace, secondary = NUTS.
           _sanity_check(primary_idata, "laplace")
           _sanity_check(secondary_idata, "blackjax")
       else:
           raise ValueError(
               f"method must be one of "
               f"{{'blackjax', 'laplace', 'both'}}, got {method!r}"
           )
   ```
