```python
def _build_session_scanner_patrl(model_name: str):
    """Build (base_attrs, scan_fn) once for the PAT-RL Network topology.

    Parameters
    ----------
    model_name : {"hgf_2level_patrl", "hgf_3level_patrl"}

    Returns
    -------
    tuple
        ``(base_attrs, scan_fn, belief_node_idx)`` suitable for lax.scan use.
    """
    if model_name == "hgf_2level_patrl":
        net = build_2level_network_patrl()
        belief_idx = 1
    elif model_name == "hgf_3level_patrl":
        net = build_3level_network_patrl()
        belief_idx = 1  # continuous value node
    else:
        raise ValueError(
            f"model_name must be 'hgf_2level_patrl' or 'hgf_3level_patrl', "
            f"got {model_name!r}"
        )
    # Prime scan_fn via one dummy input.  Shape: (1, 1) binary input.
    dummy = np.zeros((1, 1))
    net.input_data(input_data=dummy, time_steps=np.ones((1, 1)))
    base_attrs = net.attributes
    scan_fn = net.scan_fn
    return base_attrs, scan_fn, belief_idx
```

```python
def build_logp_fn_batched_patrl(
    state_arr: np.ndarray,           # (P, n_trials) int {0,1}
    choices_arr: np.ndarray,         # (P, n_trials) int {0,1}: 1=approach
    reward_mag_arr: np.ndarray,      # (P, n_trials) float
    shock_mag_arr: np.ndarray,       # (P, n_trials) float
    trial_mask: np.ndarray,          # (P, n_trials) bool
    model_name: str,
):
    """Build a batched (vmap'd) JAX logp function for PAT-RL Model A.

    Returns a pure callable that takes a params dict and returns a scalar
    summed logp across all participants.
    """
    base_attrs, scan_fn, belief_idx = _build_session_scanner_patrl(model_name)
    n_trials = state_arr.shape[1]

    def _single_logp(params: dict, state, choices, reward, shock, mask):
        # 1) Install omega_2 (and omega_3/kappa for 3-level) into base_attrs copy.
        attrs = {**base_attrs}
        attrs[belief_idx] = {**attrs[belief_idx],
                             "tonic_volatility": params["omega_2"]}
        if model_name == "hgf_3level_patrl":
            attrs[2] = {**attrs[2],
                        "tonic_volatility": params["omega_3"],
                        "volatility_coupling_children": (params["kappa"],),
                        "mean": params["mu3_0"]}
        # 2) Build scan inputs: state as binary input, ones for time_steps.
        values = state.astype(jnp.float32)[:, None]         # (T, 1)
        observed = jnp.ones((n_trials, 1), dtype=jnp.int32)  # always observed
        time_steps = jnp.ones((n_trials, 1), dtype=jnp.float32)
        rng_keys = None
        # 3) Scan with tapas-style Layer-2 clamping.  _MU_2_BOUND = 14.0.
        #    Re-implement clamp inline — do NOT import the pick_best_cue
        #    _clamped_step (that keeps the parallel-stack invariant intact).
        _MU_2_BOUND = 14.0

        def _clamped_step(carry, scan_input):
            value_i, observed_i, time_step_i = scan_input
            new_carry = scan_fn(
                carry,
                (value_i, observed_i, time_step_i),
            )
            # Layer-2 clamp: if |mean| exceeds bound, revert to prior state.
            new_mean = new_carry[belief_idx]["mean"]
            is_stable = jnp.all(jnp.isfinite(new_mean)) & (
                jnp.abs(new_mean) < _MU_2_BOUND
            )
            safe_carry = jax.tree_util.tree_map(
                lambda new, old: jnp.where(is_stable, new, old),
                new_carry,
                carry,
            )
            # Per-trial output = belief-node mean (mu2) after clamp.
            return safe_carry, safe_carry[belief_idx]["mean"]

        # 4) Scan: carry = attrs pytree; scan over trials; collect mu2 per step.
        _, mu2_traj = jax.lax.scan(
            _clamped_step,
            attrs,
            (values, observed, time_steps),
        )
        # mu2_traj shape: (T,) — scalar mean per trial.

        # 5) Compute per-trial Model A log-likelihood.
        logp_per_trial = model_a_logp(mu2_traj, choices, reward, shock,
                                       params["beta"])
        # 6) Apply trial mask; sum.
        return jnp.sum(jnp.where(mask, logp_per_trial, 0.0))

    _batched = jax.vmap(
        _single_logp,
        in_axes=(None, 0, 0, 0, 0, 0),
    )

    def logp_fn(params: dict) -> jnp.ndarray:
        # params contains per-participant arrays, shape (P,) for each.
        per_participant = jax.vmap(
            lambda p: _single_logp({k: v for k, v in zip(params.keys(),
                                                          [params[k][p] for k in params])},
                                   state_arr[p], choices_arr[p],
                                   reward_mag_arr[p], shock_mag_arr[p],
                                   trial_mask[p]),
        )(jnp.arange(state_arr.shape[0]))
        return jnp.sum(per_participant)

    return logp_fn
```

(The final shape of the vmap wiring can follow whatever pattern
`hierarchical.py::build_logp_fn_batched` uses — copy the structural template,

invariant is: the factory calls `_build_session_scanner_patrl` ONCE outside
vmap/jit, then `_single_logp` uses the closed-over `scan_fn` inside.)
